        const zhFsLabel = document.getElementById('zhFsLabel');

        const zhOpacityInput = document.getElementById('zhOpacityInput');

        const marginInput = document.getElementById('marginInput');

        

        const bgOpacityInput = document.getElementById('bgOpacityInput');

        const bgOpacityLabel = document.getElementById('bgOpacityLabel');

        const bgPaddingInput = document.getElementById('bgPaddingInput');

        const bgPaddingLabel = document.getElementById('bgPaddingLabel');

        const bgColorHex = document.getElementById('bgColorHex');

        const subBgContainer = document.getElementById('subBgContainer');

        const bgPreviewDot = document.getElementById('bgPreviewDot');


        const wordHex = document.getElementById('wordHex');

        const collocHex = document.getElementById('collocHex');

        const sentHex = document.getElementById('sentHex');


        const textInput = document.getElementById('textInput');

        const translationInput = document.getElementById('translationInput');

        const mainSub = document.getElementById('mainSub');

        const subSub = document.getElementById('subSub');

        const assOverlay = document.getElementById('assOverlay');


        function updateColors() {

            root.style.setProperty('--word-color', wordHex.value);

            root.style.setProperty('--collocation-color', collocHex.value);

            root.style.setProperty('--sentence-color', sentHex.value);

            document.getElementById('wordDot').style.backgroundColor = wordHex.value;

            document.getElementById('collocDot').style.backgroundColor = collocHex.value;

            document.getElementById('sentDot').style.backgroundColor = sentHex.value;

        }


        [wordHex, collocHex, sentHex].forEach(input => input.addEventListener('input', updateColors));


        function updateSubBackground() {

            const hex = bgColorHex.value;

            const alpha = bgOpacityInput.value / 100;

            const padding = bgPaddingInput.value + 'px';

            const r = parseInt(hex.slice(1, 3), 16);

            const g = parseInt(hex.slice(3, 5), 16);

            const b = parseInt(hex.slice(5, 7), 16);

            const rgbaColor = `rgba(${r}, ${g}, ${b}, ${alpha})`;

            subBgContainer.style.backgroundColor = rgbaColor;

            subBgContainer.style.padding = padding;

            bgPreviewDot.style.backgroundColor = hex;

            bgOpacityLabel.innerText = bgOpacityInput.value + '%';

            bgPaddingLabel.innerText = padding;

        }


        [bgColorHex, bgOpacityInput, bgPaddingInput].forEach(el => el.addEventListener('input', updateSubBackground));


        fsInput.addEventListener('input', (e) => {

            fsLabel.innerText = e.target.value + 'px';

            mainSub.style.fontSize = e.target.value + 'px';

        });

        zhFsInput.addEventListener('input', (e) => {

            zhFsLabel.innerText = e.target.value + 'px';

            subSub.style.fontSize = e.target.value + 'px';

        });

        zhOpacityInput.addEventListener('input', (e) => {

            subSub.style.opacity = e.target.value / 100;

        });

        marginInput.addEventListener('input', (e) => {

            assOverlay.style.bottom = e.target.value + '%';

        });


        textInput.addEventListener('input', () => { mainSub.innerHTML = textInput.innerHTML; });

        translationInput.addEventListener('input', () => { subSub.innerHTML = translationInput.innerHTML; });


        function wrapTag(type) {

            const sel = window.getSelection();

            if (!sel.rangeCount || sel.toString().length === 0) return;

            let container = sel.anchorNode.parentElement;

            while(container && container.id !== 'textInput' && container.id !== 'translationInput') {

                container = container.parentElement;

            }

            if(!container) return;

            const range = sel.getRangeAt(0);

            let className = type === 'word' ? 'highlight-word' : (type === 'colloc' ? 'highlight-collocation' : 'highlight-sentence');

            const span = document.createElement('span');

            span.className = className;

            span.textContent = sel.toString();

            range.deleteContents();

            range.insertNode(span);

            mainSub.innerHTML = textInput.innerHTML;

            subSub.innerHTML = translationInput.innerHTML;

        }


        updateSubBackground();

        updateColors();
